const http = require("http"); const fs = require("fs"); //Reading html: { "content-type": "text/html" } //Reading json: { "content-type": "application/json" } //Reading png: { "content-type": "image/png" } http.createServer((req, res) => { const readStream = fs.createReadStream("./static/index.html"); //file path res.writeHead(200, { "content-type": "text/html" }); readStream.pipe(res); }) .listen(3000);
